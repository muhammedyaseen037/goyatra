// Scroll Animation
const fades = document.querySelectorAll('.fade');

window.addEventListener('scroll', () => {
  fades.forEach(f => {
    const top = f.getBoundingClientRect().top;
    if(top < window.innerHeight - 100){
      f.classList.add('show');
    }
  });
});
